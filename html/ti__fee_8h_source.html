252/* Structure used when defining virtual sectors */
253/* The following error checks need to be performed: */
254/* Virtual Sector definitions are not allowed to overlap */
255/* Virtual Sector definition is at least twice the size in bytes of the total size of all defined blocks */
256/* We will need to define a formula to indicate if the number of write cycles indicated in the block definitions */
257/* is possible in the defined Virtual Sector. */
258/* Ending sector cannot be less than Starting sector */
259typedef struct
260{
261 uint16 FeeVirtualSectorNumber; /* Virtual Sector's Number - 0 and 0xFFFF values are not allowed*/
262 /* Minimum 1, Maximum 4 */
263 uint16 FeeFlashBank; /* Flash Bank to use for virtual sector. */
264 /* As we do not allow Flash EEPROM Emulation in Bank 0,
265 0 is not a valid option */
266 /* Defaultvalue 1, Minimum 1, Maxiumum 7 */
267 Fapi_FlashSectorType FeeStartSector; /* Defines the Starting Sector inthe Bank for this VirtualSector*/
268 Fapi_FlashSectorType FeeEndSector; /* Defines the Ending Sector inthe Bank for this Virtual Sector */
269 /* Start and End sectors can be the same, which indicates only
270 one sector */
271 /* is the entire virtual sector. */
272 /* Values are based on the FLASH_SECT enum */
273 /* Defaultvalue and Min is the same sector defined as the starting
274 sector */
275 /* Max values are based onthe device definition file being used.*/
276} Fee_VirtualSectorConfigType;
